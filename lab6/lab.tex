}
\btext{.5}{.1}{.45}{Adds a Tree Node by:\\
\qi{Checking if the element in array table is a dot or dash}

\normalpage{Decode}{
\VerbatimInput{decodeBook.cpp}
}
\btext{.5}{.1}{.45}{Converts morse code to English by:\\
\qi{Looking at each character of the message}
\qi{If it is a dot, then it moves down the left branch}
\qi{It it is a dash, then it moves down the right branch}
\qi{If it is a space, then it finds the current Tree node's symbol
and adds that to the text while adding a space and returning to root}
\qi{Because of this when it encounters a space between words it adds
two spaces since the current node at space will be root whose symbol
is a space}
}
